	node wcc/base/inline-templates.js

    cd wcc/WccSkills

переделал метод _loadProjects в WccPortfolio.js .
Теперь он сразу присваивает переменной this._projects промис (Promise), который резолвится данными.
Раньше this._projects был undefined до момента завершения fetch , 
и если рендер начинался раньше, чем приходил ответ от сервера, возникала ошибка или пустой список.
  _loadProjects() {
    this._projects = fetch("projects.json")
      .then(response => {
        if (!response.ok) throw new Error('impossible');
        return response.json();
      })
      .catch(e => {
        console.warn(e.message);
        return [];
      });
  }
